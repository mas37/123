    def __init__(self, filename, atoms_list=None):
        """Init checkpoint

        Args:
            filename: eg: model.ckpt-0
            atom_list: atom sequence eg ['H','C','N','O'],
                       if None atomic species can not be inferred by name
                       just by position
        """
        super().__init__()
        self.filename = filename
        self.step = int(filename.split('-')[1].split('.')[0])
        self._atoms_list = atoms_list
        if atoms_list:
            self._atoms_dict = {n: i for i, n in enumerate(atoms_list)}
        self._load_file()

    def _load_file(self):
        """Internal file loader
        """
        reader = pywrap_tensorflow.NewCheckpointReader(self.filename)
        var_to_shape_map = reader.get_variable_to_shape_map()
        tensors_value = {}

        for key in var_to_shape_map:
            tensors_value[key] = reader.get_tensor(key)

        self._tensors_value = tensors_value
        self._tensors_shape = var_to_shape_map

    def _recover_network_structure(self):
        """ recover the network structure using the labels of the tensors

        Return:
            architecture:
            [(species_idx, depth), (species_idx, depth), .....]
        """
        network_elements = []
        for layer in self._tensors_shape.keys():
            matches = re.findall(r'^species_(\d+)_layer_(\d+)', layer)
            if len(matches) == 1:
                network_elements.append(tuple([int(x) for x in matches[0]]))
        network_elements_matrix = np.asarray(list(set(network_elements)))
        n_species = network_elements_matrix.max(axis=0)[0] + 1
        architecture = []
        for species in range(n_species):
            loc = np.where(network_elements_matrix[:, 0] == species)
            architecture.append(
                network_elements_matrix[loc].max(axis=0).tolist())
        return architecture

    def dump_checkpoint_to_single_files(self,
                                        directory='dump_checkpoint',
                                        dumping_function=np.save,
                                        dumping_extension='npy'):
        """Dump weights and biases of a checkpoint to a directory, also metadata

        Args:
            directory: where to dump the files
            dumping_function: the function that will be used to dump the data,
                              default numpy array arguments (file, arr)
            dumping_extension: the extension of the dumped files

        Return:
            None
        """
        if not os.path.exists(directory):
            os.makedirs(directory)
        networks_sructure = self._recover_network_structure()
        networks_files = []
        networks_layers_sizes = []
        for specie, depth in networks_sructure:
            specie_network_files = []
            specie_network_layers_sizes = []
            if self._atoms_list:
                specie_name = self._atoms_list[specie]
            else:
                specie_name = specie
            for layer in range(1, depth + 1):
                base_name = 'species_{}_layer_{}/'.format(specie, layer)

                w_name = base_name + 'weights'
                b_name = base_name + 'biases'
                w_size = self._tensors_shape[w_name]
                b_size = self._tensors_shape[b_name]
                w_file_name = 'species_{}_layer_{}_weights_{}x{}.{extension}'.format(
                    specie_name, layer, *w_size, extension=dumping_extension)
                b_file_name = 'species_{}_layer_{}_biases_{}.{extension}'.format(
                    specie_name, layer, *b_size, extension=dumping_extension)

                specie_network_files.append((w_file_name, b_file_name))
                specie_network_layers_sizes.append((w_size, b_size))

                dumping_function(
                    os.path.join(directory, w_file_name),
                    self._tensors_value[w_name])
                dumping_function(
                    os.path.join(directory, b_file_name),
                    self._tensors_value[b_name])

            networks_files.append(specie_network_files)
            networks_layers_sizes.append(specie_network_layers_sizes)

        metadata = {
            'networks_structure': networks_sructure,
            'networks_files': networks_files,
            'networks_layers_sizes': networks_layers_sizes
        }
        if self._atoms_list:
            metadata['atoms_list'] = self._atoms_list
        with open(os.path.join(directory, 'networks_metadata.json'), 'w') as f:
            json.dump(metadata, f)
        return None

    def norms_calculator(self):
        """Calculate the norm 1 and 2 of all the tensors

        Return:
            a dict {tensor_name:(l1, l2)}

        l1 = sum_i(abs(v_i))
        l2 = sum_i(v_i**2)/2
        """
        n = {}
        for k, v in self._tensors_value.items():
            n[k] = (np.sum(np.abs(v)), np.sum(v**2) / 2)
        return n

    def convert_names(self, change_dictionay):
        """Variables name changer

        Args:
            change_dictionay: format {'old_variable_name':'new_variable_name'}

        Return:
            List of not_found conversion

        Useful when moving variables to new graph structure
        """
        not_found = []
        for old_key in change_dictionay:
            old_keys = self._tensors_value.keys()
            if old_key in old_keys:
                new_key = change_dictionay[old_key]
                self._tensors_value[new_key] = self._tensors_value.pop(old_key)
                self._tensors_shape[new_key] = self._tensors_shape.pop(old_key)
            else:
                not_found.append(old_key)
        return not_found

    def _convert_kind_to_idx(self, atom_kind):
        """Convert an atom kind to it's index

        Args:
            atom_name: string with atom kind

        Return:
            integer corresponding to idx of that atom
        """

        if not self._atoms_list:
            raise ValueError('atoms list not supplied, method not\
                              available')
        return self._atoms_dict[atom_kind]

    def substitute_species_with_species(self, idx1, idx2):
        """Substitute s1 with s2

        Args:
            s1:species 1, index or kind,
               kind available only if atoms list was supplied
            s2:species 2, index of kind,
               kind available only if atoms list was supplied
        Return:
            None

        note: type check should be not done, but for now annotations are not
              available
        """
        if isinstance(idx1, str):
            idx1 = self._convert_kind_to_idx(idx1)
        if isinstance(idx2, str):
            idx2 = self._convert_kind_to_idx(idx2)

        for k, v in self._tensors_value.items():
            try:
                v[idx1] = v[idx2]
            except IndexError as e:
                if 'invalid index to scalar variable' in e.args[0]:
                    continue
                else:
                    raise e

            species_network = []

                species_network.append((self._tensors_value[w_name],
                                        self._tensors_value[b_name]))
            networks.append(species_network)
        return networks

    @classmethod
    def ckpts_analizer(cls, ckpt):
        """Analyze checkpoint file.

        Args:
            ckpt: checkpoint file

        Return:
            list of steps where ckpts were created.

        Rise:
            FileNotFoundError: if file not found
            Value Error: if the checkpoint file is incoherent
        """
        ckpts = []
        try:
            with open(ckpt) as f:
                line0 = f.readline()
                line0 = line0.split('"')[1].split('-')[1]
                for x in f:
                    xt = x.split('"')[1].split('-')[1]
                    ckpts.append(int(xt))
            ckpts.sort()
            if ckpts[-1] != int(line0):
                raise ValueError('error on last line')
            return ckpts
        except FileNotFoundError as e:
            raise e

    @classmethod
    def files_finder(cls, directory):
        return [
            os.path.join(directory, 'model.ckpt-' + str(x))
            for x in Checkpoint.steps_finder(directory)
        ]

    """ A class that load parameters form files generated by  and is compatible
    with the Network object
    """

    def __init__(self, file_name, atoms_list=None):
        """init Parameters

        Args:
            file_name: file_name and path of the network file
                      network file is a json, can be manually crafted
            atom_list: atom sequence eg ['H','C','N','O'],
                       if None atomic species can not be inferred by name
                       just by position
                       gets over witten by file if file has one
        """

        super().__init__()
        self._file_name = file_name
        self._path = os.path.dirname(file_name)
        self._atoms_list = atoms_list
        self.step = 0
        self._load_file()

    def _load_file(self):
        with open(self._file_name) as f:
            data = json.load(f)
        self._networks_structure = data['networks_structure']
        self._networks_files = data['networks_files']
        self._networks_layers_sizes = data['networks_layers_sizes']
        self._atoms_list = data.get('atoms_list', self._atoms_list)

    def consistency_check(self, load_function=np.load):
        """ check consistency of the shape saved inside the json
            with those exposed by files

        Args:
            load function, the resulting object must have shape proprieties

        Return:
            True if consistency is passed,
            False otherwise
        """
        for species_network_files, specie_network_layers_sizes in zip(
                self._networks_files, self._networks_layers_sizes):
            for files_names, shape in zip(species_network_files,
                                          specie_network_layers_sizes):
                w_file, b_file = files_names
                w = load_function(os.path.join(self._path, w_file))
                b = load_function(os.path.join(self._path, b_file))
                if not [list(w.shape), list(b.shape)] == shape:
                    return False
        return True

    def tensors_value_organized_as_networks_structures(self,
                                                       load_function=np.load):
        """a list of list with all the tensor organized

        Return:
            list(over species) of list(over layers) of tuples(w, b)
            w, b are numpy array float32
            [ [(w, b), (w, b), ...], [(w, b)....], ...]
        """

        networks = []
        for species_network_files in self._networks_files:
            species_network = []
            for w_file, b_file in species_network_files:
                w = load_function(os.path.join(self._path, w_file))
                b = load_function(os.path.join(self._path, b_file))
                species_network.append((w, b))
            networks.append(species_network)
        return networks
